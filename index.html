-- ════════════════════════════════════════════════════════════════════════════ -- Selebox — Phase 6: book lock + video monetization (time-based) -- -- New rules (April 2026): -- -- Books — book-level lock -- Author sets a single integer "lock_from_chapter" on the book. -- All chapters with chapter_number >= lock_from_chapter are locked. -- Chapters 1 to (lock_from_chapter - 1) are always free. -- lock_from_chapter must be >= 5 (chapters 1-4 always free). -- Once locked, the lock cannot be REMOVED (set to null) for 90 days. -- Author MAY change the value (5 -> 8, 5 -> 10, etc.) within 90 days -- because that only changes which chapters are locked, not the fact of -- monetization. Setting back to null is what's restricted. -- -- Whole-book unlock -- Reader can unlock all currently-locked chapters in one transaction -- for a discounted price (default 15% off). Discount lives in -- app_config.book_bulk_unlock_discount_pct (admin-editable). -- Already-unlocked chapters are excluded from the bulk price. -- -- Videos — time-based monetization -- Author flags the video as monetized (videos.is_monetized). -- Free for the first 3 minutes (app_config.video_initial_unlock_seconds). -- At the 3:00 mark, viewer chooses: -- Coin path → 1 coin → permanent unlock (unlocks row written) -- Star path → 1 star → continues, charged again at 13:00, 23:00, ... -- every video_recurring_unlock_seconds (default 600). -- Re-watching below the high-water mark never re-charges. -- -- Idempotent — safe to re-run. -- ════════════════════════════════════════════════════════════════════════════ -- ── 1. Schema additions ──────────────────────────────────────────────── alter table public.books add column if not exists lock_from_chapter int, add column if not exists locked_at timestamptz; -- chapter_number floor: 5 (chapters 1-4 always free). Allow null = unlocked. alter table public.books drop constraint if exists books_lock_from_chapter_check; alter table public.books add constraint books_lock_from_chapter_check check (lock_from_chapter is null or lock_from_chapter >= 5); create index if not exists books_locked_idx on public.books(lock_from_chapter) where lock_from_chapter is not null; alter table public.videos add column if not exists is_monetized boolean not null default false; -- ── 2. video_progress (high-water mark, prevents double-charging on rewatch) ── create table if not exists public.video_progress ( user_id uuid not null references public.profiles(id) on update cascade on delete cascade, video_id uuid not null references public.videos(id) on delete cascade, max_position_seconds int not null default 0 check (max_position_seconds >= 0), -- "Paid through" — the user has paid all star thresholds up to (and including) -- this position. We use this so the player never re-prompts for thresholds -- the user already passed. Coin payment sets this to a very high value -- (effectively forever). paid_through_seconds int not null default 0 check (paid_through_seconds >= 0), updated_at timestamptz not null default now(), primary key (user_id, video_id) ); alter table public.video_progress enable row level security; drop policy if exists "video_progress read self or mod" on public.video_progress; create policy "video_progress read self or mod" on public.video_progress for select using (auth.uid() = user_id or public.is_moderator()); -- No direct write — all updates flow through SECURITY DEFINER RPCs below. -- ── 3. app_config additions for the new defaults ────────────────────── insert into public.app_config (key, value_int, description) values ('book_bulk_unlock_discount_pct', 15, 'Discount percentage applied when a reader unlocks all locked chapters in one transaction'), ('video_initial_unlock_seconds', 180, 'Seconds before the first unlock prompt on monetized videos (default 3 minutes)'), ('video_recurring_unlock_seconds', 600, 'Seconds between subsequent star deductions on monetized videos (default 10 minutes)') on conflict (key) do nothing; -- ── 4. Expand transaction-type CHECK to allow new types ────────────── alter table public.coin_transactions drop constraint if exists coin_transactions_type_check; alter table public.coin_transactions add constraint coin_transactions_type_check check (type in ( 'purchase', 'unlock_video', 'unlock_chapter', 'unlock_book_bulk', 'admin_adjust', 'refund', 'migration_grant', 'tip_sent', 'tip_received' )); alter table public.star_transactions drop constraint if exists star_transactions_type_check; alter table public.star_transactions add constraint star_transactions_type_check check (type in ( 'ad_watch', 'unlock_video', 'unlock_chapter', 'unlock_book_bulk', 'admin_adjust', 'expire', 'migration_grant' )); -- ── 5. 90-day-no-removal trigger on books.lock_from_chapter ────────── create or replace function public.enforce_book_lock_grace_period() returns trigger language plpgsql as $$ begin -- First-time lock → stamp locked_at, allow. if old.lock_from_chapter is null and new.lock_from_chapter is not null then new.locked_at := now(); return new; end if; -- Removing the lock entirely — only allowed after 90 days from initial lock. if old.lock_from_chapter is not null and new.lock_from_chapter is null then if old.locked_at is not null and old.locked_at > now() - interval '90 days' then raise exception 'Cannot remove monetization within 90 days of locking. Lock was set on %, eligible for removal on %.', to_char(old.locked_at, 'YYYY-MM-DD'), to_char(old.locked_at + interval '90 days', 'YYYY-MM-DD') using errcode = 'P0001'; end if; new.locked_at := null; return new; end if; -- Changing the value while still locked (e.g. 5 -> 8) is allowed any time. -- DO NOT update locked_at — the 90-day window is anchored to the FIRST lock, -- not subsequent edits. return new; end; $$; drop trigger if exists trg_book_lock_grace on public.books; create trigger trg_book_lock_grace before update on public.books for each row when (old.lock_from_chapter is distinct from new.lock_from_chapter) execute function public.enforce_book_lock_grace_period(); -- ── 6. RPC: unlock_book_bulk ───────────────────────────────────────── create or replace function public.unlock_book_bulk( p_book_id uuid, p_currency text -- 'coin' | 'star' ) returns jsonb language plpgsql security definer set search_path = public as $$ declare v_user_id uuid := auth.uid(); v_book public.books%rowtype; v_locked_count int; v_per_chapter int; v_total_before int; v_discount_pct int; v_total_after int; v_balance_after int; v_chapter_id uuid; begin if v_user_id is null then return jsonb_build_object('ok', false, 'error', 'not_authenticated'); end if; if p_currency not in ('coin', 'star') then return jsonb_build_object('ok', false, 'error', 'invalid_currency'); end if; select * into v_book from public.books where id = p_book_id; if not found then return jsonb_build_object('ok', false, 'error', 'book_not_found'); end if; if v_book.lock_from_chapter is null then return jsonb_build_object('ok', false, 'error', 'no_locks_on_book'); end if; -- Count chapters that are locked AND not yet unlocked by this user. -- "Locked" = published AND chapter_number >= book.lock_from_chapter. select count(*) into v_locked_count from public.chapters c where c.book_id = p_book_id and c.is_published = true and c.chapter_number >= v_book.lock_from_chapter and not exists ( select 1 from public.unlocks u where u.user_id = v_user_id and u.target_type = 'chapter' and u.target_id = c.id::text ); if v_locked_count = 0 then return jsonb_build_object('ok', true, 'already_unlocked', true, 'cost', 0); end if; -- Pricing: per-chapter default × count, then apply discount. v_per_chapter := public.get_config_int( case when p_currency = 'coin' then 'default_chapter_unlock_coins' else 'default_chapter_unlock_stars' end ); if v_per_chapter is null or v_per_chapter <= 0 then return jsonb_build_object('ok', false, 'error', 'cost_unresolved'); end if; v_total_before := v_per_chapter * v_locked_count; v_discount_pct := coalesce(public.get_config_int('book_bulk_unlock_discount_pct'), 15); -- Floor of 1 so we never charge zero. v_total_after := greatest(1, v_total_before - (v_total_before * v_discount_pct / 100)); -- Atomic debit (only succeeds if balance is sufficient). if p_currency = 'coin' then update public.wallets set coin_balance = coin_balance - v_total_after, updated_at = now() where user_id = v_user_id and coin_balance >= v_total_after returning coin_balance into v_balance_after; else update public.wallets set star_balance = star_balance - v_total_after, updated_at = now() where user_id = v_user_id and star_balance >= v_total_after returning star_balance into v_balance_after; end if; if v_balance_after is null then return jsonb_build_object( 'ok', false, 'error', 'insufficient_balance', 'cost', v_total_after, 'currency', p_currency, 'chapters_to_unlock', v_locked_count ); end if; -- Insert unlock rows for every still-locked chapter. for v_chapter_id in select c.id from public.chapters c where c.book_id = p_book_id and c.is_published = true and c.chapter_number >= v_book.lock_from_chapter and not exists ( select 1 from public.unlocks u where u.user_id = v_user_id and u.target_type = 'chapter' and u.target_id = c.id::text ) loop insert into public.unlocks (user_id, target_type, target_id, paid_currency, paid_amount, source) values (v_user_id, 'chapter', v_chapter_id::text, p_currency, -- Per-chapter share of the bulk price (rounded down). v_total_after / v_locked_count, 'web') on conflict (user_id, target_type, target_id) do nothing; end loop; -- Single ledger row for the whole bulk transaction. if p_currency = 'coin' then insert into public.coin_transactions (user_id, delta, balance_after, type, reference_type, reference_id, metadata) values (v_user_id, -v_total_after, v_balance_after, 'unlock_book_bulk', 'book', p_book_id::text, jsonb_build_object( 'chapters_unlocked', v_locked_count, 'cost_before_discount', v_total_before, 'discount_pct', v_discount_pct, 'per_chapter', v_per_chapter )); else insert into public.star_transactions (user_id, delta, balance_after, type, reference_type, reference_id, metadata) values (v_user_id, -v_total_after, v_balance_after, 'unlock_book_bulk', 'book', p_book_id::text, jsonb_build_object( 'chapters_unlocked', v_locked_count, 'cost_before_discount', v_total_before, 'discount_pct', v_discount_pct, 'per_chapter', v_per_chapter )); end if; return jsonb_build_object( 'ok', true, 'chapters_unlocked', v_locked_count, 'cost', v_total_after, 'cost_before_discount', v_total_before, 'discount_pct', v_discount_pct, 'currency', p_currency, 'balance_after', v_balance_after ); end; $$; grant execute on function public.unlock_book_bulk(uuid, text) to authenticated; -- ── 7. RPC: unlock_video_threshold (time-based deduction) ────────────── -- -- Called by the player when the user crosses a paid threshold. -- p_threshold_seconds = 180 (initial), 780, 1380, 1980, ... (recurring star). -- -- For 'coin': debits 1 coin once, inserts an 'unlocks' row, sets -- video_progress.paid_through_seconds to a sentinel large value (effectively -- "permanent"), writes ledger. -- -- For 'star': debits 1 star, bumps paid_through_seconds to (threshold + -- recurring_window - 1), writes ledger. No unlocks row — the user pays again -- at the next threshold. create or replace function public.unlock_video_threshold( p_video_id text, -- text to allow legacy aw_/sb_ ids p_currency text, -- 'coin' | 'star' p_threshold_seconds int -- 180, 780, 1380, ... ) returns jsonb language plpgsql security definer set search_path = public as $$ declare v_user_id uuid := auth.uid(); v_video_uuid uuid; v_legacy bool := p_video_id like 'aw\_%' or p_video_id like 'sb\_%'; v_cost int; v_balance_after int; v_recurring int := coalesce(public.get_config_int('video_recurring_unlock_seconds'), 600); v_already_paid bool := false; v_progress record; begin if v_user_id is null then return jsonb_build_object('ok', false, 'error', 'not_authenticated'); end if; if p_currency not in ('coin', 'star') then return jsonb_build_object('ok', false, 'error', 'invalid_currency'); end if; if p_threshold_seconds is null or p_threshold_seconds < 0 then return jsonb_build_object('ok', false, 'error', 'invalid_threshold'); end if; -- Resolve UUID for video_progress / unlocks rows. Legacy ids stay as text -- in unlocks but video_progress requires a UUID FK to videos. if v_legacy then -- Legacy videos don't get tracked in video_progress (no FK target). -- Treat each threshold as a fresh debit, no rewatch protection. v_video_uuid := null; else begin v_video_uuid := p_video_id::uuid; exception when others then return jsonb_build_object('ok', false, 'error', 'invalid_video_id'); end; end if; -- Already permanently unlocked via coin? Skip everything. if exists ( select 1 from public.unlocks where user_id = v_user_id and target_type = 'video' and target_id = p_video_id ) then return jsonb_build_object('ok', true, 'already_permanent', true); end if; -- Already paid through this threshold via stars? Skip the debit. if v_video_uuid is not null then select * into v_progress from public.video_progress where user_id = v_user_id and video_id = v_video_uuid; if found and v_progress.paid_through_seconds >= p_threshold_seconds then return jsonb_build_object('ok', true, 'already_paid_for_threshold', true, 'paid_through_seconds', v_progress.paid_through_seconds); end if; end if; -- Resolve cost from app_config defaults. v_cost := public.get_config_int( case when p_currency = 'coin' then 'default_video_unlock_coins' else 'default_video_unlock_stars' end ); if v_cost is null or v_cost <= 0 then return jsonb_build_object('ok', false, 'error', 'cost_unresolved'); end if; -- Atomic debit. if p_currency = 'coin' then update public.wallets set coin_balance = coin_balance - v_cost, updated_at = now() where user_id = v_user_id and coin_balance >= v_cost returning coin_balance into v_balance_after; else update public.wallets set star_balance = star_balance - v_cost, updated_at = now() where user_id = v_user_id and star_balance >= v_cost returning star_balance into v_balance_after; end if; if v_balance_after is null then return jsonb_build_object( 'ok', false, 'error', 'insufficient_balance', 'cost', v_cost, 'currency', p_currency, 'threshold_seconds', p_threshold_seconds ); end if; -- Coin path → permanent unlock + ledger. if p_currency = 'coin' then insert into public.unlocks (user_id, target_type, target_id, paid_currency, paid_amount, source) values (v_user_id, 'video', p_video_id, 'coin', v_cost, 'web') on conflict (user_id, target_type, target_id) do nothing; -- Mark video_progress paid_through to a sentinel "forever" value if v_video_uuid is not null then insert into public.video_progress (user_id, video_id, max_position_seconds, paid_through_seconds, updated_at) values (v_user_id, v_video_uuid, p_threshold_seconds, 2147483647, now()) on conflict (user_id, video_id) do update set paid_through_seconds = greatest(public.video_progress.paid_through_seconds, excluded.paid_through_seconds), max_position_seconds = greatest(public.video_progress.max_position_seconds, excluded.max_position_seconds), updated_at = now(); end if; insert into public.coin_transactions (user_id, delta, balance_after, type, reference_type, reference_id, metadata) values (v_user_id, -v_cost, v_balance_after, 'unlock_video', 'video', p_video_id, jsonb_build_object('threshold_seconds', p_threshold_seconds, 'mode', 'permanent')); return jsonb_build_object( 'ok', true, 'mode', 'permanent', 'cost', v_cost, 'currency', 'coin', 'balance_after', v_balance_after ); end if; -- Star path → bump paid_through, NO unlocks row, ledger as 'unlock_video' with mode=window. if v_video_uuid is not null then insert into public.video_progress (user_id, video_id, max_position_seconds, paid_through_seconds, updated_at) values (v_user_id, v_video_uuid, greatest(p_threshold_seconds, p_threshold_seconds + v_recurring - 1), p_threshold_seconds + v_recurring - 1, now()) on conflict (user_id, video_id) do update set paid_through_seconds = greatest(public.video_progress.paid_through_seconds, excluded.paid_through_seconds), max_position_seconds = greatest(public.video_progress.max_position_seconds, excluded.max_position_seconds), updated_at = now(); end if; insert into public.star_transactions (user_id, delta, balance_after, type, reference_type, reference_id, metadata) values (v_user_id, -v_cost, v_balance_after, 'unlock_video', 'video', p_video_id, jsonb_build_object('threshold_seconds', p_threshold_seconds, 'mode', 'window', 'paid_through_seconds', p_threshold_seconds + v_recurring - 1)); return jsonb_build_object( 'ok', true, 'mode', 'window', 'cost', v_cost, 'currency', 'star', 'balance_after', v_balance_after, 'paid_through_seconds', p_threshold_seconds + v_recurring - 1 ); end; $$; grant execute on function public.unlock_video_threshold(text, text, int) to authenticated; -- ── 8. RPC: bump_video_progress (no payment, just track high-water mark) ── -- -- Called by the player periodically while the user is in a free zone, so that -- subsequent sessions know "they've already watched up to here". Doesn't move -- paid_through_seconds — only max_position_seconds. create or replace function public.bump_video_progress( p_video_id uuid, p_position_seconds int ) returns void language plpgsql security definer set search_path = public as $$ declare v_user_id uuid := auth.uid(); begin if v_user_id is null or p_position_seconds is null or p_position_seconds < 0 then return; end if; insert into public.video_progress (user_id, video_id, max_position_seconds, paid_through_seconds, updated_at) values (v_user_id, p_video_id, p_position_seconds, 0, now()) on conflict (user_id, video_id) do update set max_position_seconds = greatest(public.video_progress.max_position_seconds, excluded.max_position_seconds), updated_at = now(); end; $$; grant execute on function public.bump_video_progress(uuid, int) to authenticated; -- ── 9.
